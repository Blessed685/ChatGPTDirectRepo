fun formatAnswer(answ: Double): String {
    var answer = ""
    if (answ % 1.0 == 0.0){
        answer = answ.toInt().toString()
    }
    else{
        answer = answ.toString()
    }
    return answer
}
@Composable
fun ThisCalculatorZeroLgc() {
    var input by remember {
        mutableStateOf("")
    }
    var resultDisplayed by remember {
        mutableStateOf(false)
    }
    Column {
        ThisScreenUi(viewMyInput = input)
        ThisKeypadUi(userInput = input, keypadPressed = { key ->

            if (key == "C"){
                input = ""
            }
            else if (key == "CE"){
                input = input.dropLast(1)
            }
            else if (key == "="){
                try {
                    if(input.contains("+")){
                        val numbers = input.split("+")
                        val operationLambda = {a: Double, b: Double -> a+b}
                        val answer = operationLambda(numbers[0].toDouble(), numbers[1].toDouble())
                        input = formatAnswer(answer)
                        resultDisplayed = true
                    }
                    else if (input.contains("-")){
                        val numbers = input.split("-")
                        val operationLambda = {a: Double, b: Double -> a-b}
                        val answer = operationLambda(numbers[0].toDouble(), numbers[1].toDouble())
                        input = formatAnswer(answer)
                        resultDisplayed = true
                    }
                    else if (input.contains("x")){
                        val numbers = input.split("x")
                        val operationLambda = {a: Double, b: Double -> a*b}
                        val answer = operationLambda(numbers[0].toDouble(), numbers[1].toDouble())
                        input = formatAnswer(answer)
                        resultDisplayed = true
                    }
                    else if (input.contains("/")){
                        val numbers = input.split("/")
                        val operationLambda = {a: Double, b: Double -> a/b}
                        val answer = operationLambda(numbers[0].toDouble(), numbers[1].toDouble())
                        val numbersTwo = numbers[1].toDouble()
                        if (numbersTwo == 0.0){
                            input = "Error"
                        }
                        else {
                            input = formatAnswer(answer)
                            resultDisplayed = true
                        }
                    }
                }
                catch (e: Exception){
                    input = "Error"
                }

            }
            else if (input == "Error"){
                input = key
            }
            else if (resultDisplayed){
                val operators = listOf("-", "+", "x", "/")
                if (!operators.contains(key)){
                    input = key
                    resultDisplayed = false
                }
                else if (operators.contains(key)){
                    input += key
                    resultDisplayed = false
                }
            }
            else if (key == "."){
                val operators = listOf("-", "+", "x", "/")
                if (input.isEmpty()){
                    input = "0."
                }
                else if (operators.contains(input.lastOrNull()?.toString())){
                    input += "0."
                }
                else{
                    var currentNumber = ""
                    for (character in input.reversed()){
                        if (operators.contains(character.toString())){
                            break
                        }
                        currentNumber += character
                    }
                    if (!currentNumber.contains(".")){
                        input += key
                    }
                }
            }
            else if (key == "0"){
                val operators = listOf("-", "+", "x", "/")
                var currentNumber = ""
                for (character in input.reversed()){
                    if (operators.contains(character.toString())){
                        break
                    }
                    currentNumber += character
                }
//                if (currentNumber.lastOrNull()?.toString() != "0")
                if (currentNumber != "0"){
                    input += key
                }
            }
            else {
                val operators = listOf("-", "+", "x", "/")
                if (input.isEmpty() && operators.contains(key)){

                }
                else if (operators.contains(input.lastOrNull()?.toString()) && operators.contains(key)){
                    input = input.dropLast(1) + key
                }
                else if (input.lastOrNull()?.toString() == "." && operators.contains(key)){

                }
                /*else if (resultDisplayed && !operators.contains(key)){
                    input = key
                    resultDisplayed = false
                }*/
                else {
                    input += key
                }
            }
        })
    }
}
